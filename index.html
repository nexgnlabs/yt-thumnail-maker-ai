const { Client, RemoteAuth } = require('whatsapp-web.js'); const qrcode = require('qrcode-terminal'); const axios = require('axios'); // Using Environment Variables for safety const GEMINI_API_KEY = process.env.GEMINI_KEY; const client = new Client({ authStrategy: new RemoteAuth({ /* Optional: for cloud saving */ }), puppeteer: { headless: true, args: ['--no-sandbox', '--disable-setuid-sandbox'] // Required for Linux/Cloud hosting } }); client.on('qr', (qr) => { // On GitHub/Cloud, you check the logs to see this QR qrcode.generate(qr, { small: true }); }); client.on('ready', () => { console.log('WA Bot is Online!'); }); client.on('message', async (msg) => { if (msg.body.startsWith('!ai ')) { try { const userQuery = msg.body.replace('!ai ', ''); const response = await axios.post( `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.0-flash:generateContent?key=${GEMINI_API_KEY}`, { contents: [{ parts: [{ text: userQuery }] }] } ); msg.reply(response.data.candidates[0].content.parts[0].text); } catch (err) { msg.reply("Error connecting to Gemini."); } } }); client.initialize();
